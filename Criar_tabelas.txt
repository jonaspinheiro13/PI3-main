	A PAGINA employee.html 
















CREATE TABLE entregas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome_cliente VARCHAR(100),
    numero_pedido VARCHAR(50),
    data_entrega DATE,
    hora TIME,
    valor DECIMAL(10,2),
    endereco_cliente VARCHAR(255),
    telefone VARCHAR(20),
    itens_pedido TEXT
);
CREATE TABLE entregadores (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100),
    sobrenome VARCHAR(100),
    imagem VARCHAR(255),
    dia INT,
    mes INT,
    ano INT,
    rg VARCHAR(20),
    cpf VARCHAR(20),
    rua VARCHAR(255),
    numero VARCHAR(10),
    bairro VARCHAR(100),
    estado VARCHAR(50),
    cidade VARCHAR(100),
    cep VARCHAR(20),
);
